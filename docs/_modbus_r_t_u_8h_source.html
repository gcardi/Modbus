313template<typename OutputIterator>
314OutputIterator RTUProtocol::Write( OutputIterator Out, uint16_t Data )
315{
316 *Out++ = ( Data >> 8 ) & 0xFF; // Data Hi
317 *Out++ = Data & 0xFF; // Data Lo
318 return Out;
319}

322template<typename InputIterator>
323InputIterator RTUProtocol::Read( InputIterator In, uint16_t& Data )
324{
325 uint16_t const HiData = static_cast<uint16_t>( *In++ & 0xFF ) << 8;
326 Data = HiData + *In++;
327 return In;
328}

341template<typename OutputIterator, typename InputIterator>
342OutputIterator RTUProtocol::WriteCRC( OutputIterator Out,
343 InputIterator Begin,
344 InputIterator End )
345{
346 uint16_t const CRC = ComputeCRC( Begin, End );
347 *Out++ = CRC & 0xFF;
348 *Out++ = ( CRC >> 8 ) & 0xFF;
349 return Out;
350}

353template<typename InputIterator>
354uint16_t RTUProtocol::ComputeCRC( InputIterator Begin, InputIterator End )
355{
356 return std::for_each( Begin, End, boost::crc_16_type( 0xFFFF ) ).checksum();
357}

385template<typename OutputIterator>
386bool RTUProtocol::SendAndReceiveFramesInt( Context const & Context,
387 const FrameCont& TxFrame, OutputIterator Out,
388 FrameCont::size_type RxFramelength,
389 bool NoThrow )
390{
391 static const FrameCont::size_type EatEchoExtraCharCount = 0;
392
393 FrameCont RxFrame;
394
395 RxFrame.reserve( 200 );
396
397#if defined ( _DEBUG ) && defined( VIEW_RTU_PROTOCOL_DIAG )
398 ShowBuffer(
399 Format(
400 _D( "TX(%s): " ), ARRAYOFCONST( ( commPort_.GetCommPort().c_str() ) )
401 ),
402 TxFrame.begin(), TxFrame.end()
403 );
404#endif
405 if ( onFlowEvent_ )
406 onFlowEvent_( *this, FlowDirection::TX, TxFrame );
407
408 commPort_.PurgeCommPort();
409
410 commPort_.WriteBuffer(
411 const_cast<FrameCont::value_type*>( &TxFrame[0] ), TxFrame.size()
412 );
413
414 if ( CancelTXEcho ) {
415 for ( FrameCont::size_type Cnt = TxFrame.size() + EatEchoExtraCharCount ; Cnt ; --Cnt ) {
416 uint8_t Char;
417
418 if ( !commPort_.ReadBytes( &Char, 1 ) ) {
419 if ( NoThrow ) {
420 return false;
421 }
422 else {
423 throw EContextException(
424 Context,
425 SysErrorMessage( GetLastError() )
426 );
427 }
428 }
429 }
430 }
431
432 boost::crc_16_type Crc( 0xFFFF );
433
434 for ( FrameCont::size_type Idx = 0 ; Idx < RxFramelength ; ++Idx ) {
435 uint8_t Char;
436 if ( unsigned int const BytesRead = commPort_.ReadBytes( &Char, 1 ) ) {
437 Crc.process_byte( Char );
438 RxFrame.push_back( Char );
439
440 if ( ( Char & 0x80 ) && Idx == 1 ) {
441//#if defined( _DEBUG )
442//::OutputDebugString( Format( _D( "%.2X" ), ARRAYOFCONST( ( int( Char ) ) ) ).c_str() );
443//endif
444 RxFramelength = 5;
445 }
446
447 }
448 else if ( NoThrow ) {
449 return false;
450 }
451 else if ( !BytesRead ) {
452 throw EContextException(
453 Context,
454 _D( "Timeout error" )
455 );
456 }
457 else {
458 throw EContextException(
459 Context,
460 SysErrorMessage( GetLastError() )
461 );
462 }
463 }
464
465#if defined ( _DEBUG ) && defined( VIEW_RTU_PROTOCOL_DIAG )
466 ShowBuffer(
467 Format(
468 _D( "RX(%s): " ), ARRAYOFCONST( ( commPort_.GetCommPort().c_str() ) )
469 ),
470 RxFrame.begin(), RxFrame.end()
471 );
472#endif
473
474 if ( onFlowEvent_ ) {
475 onFlowEvent_( *this, FlowDirection::RX, RxFrame );
476 }
477
478 if ( Crc.checksum() ) {
479 if ( NoThrow ) {
480 return false;
481 }
482 else {
483 throw EContextException( Context, _D( "Bad CRC (RX)" ) );
484 }
485 }
486
487 if ( RxFrame[0] != TxFrame[0] ) {
488 if ( NoThrow ) {
489 return false;
490 }
491 else {
492 throw EContextException( Context, _D( "Slave address mismatch" ) );
493 }
494 }
495
496 if ( RxFrame[1] != TxFrame[1] ) {
497 if ( NoThrow ) {
498 return false;
499 }
500 else {
501 if ( RxFrame[1] & 0x80 ) {
502 RaiseStandardException( Context, ExceptionCode( RxFrame[2] ) );
503 }
504 else {
505 throw EContextException( Context, _D( "Function code mismatch" ) );
506 }
507 }
508 }
509
510 std::copy( RxFrame.begin() + 2, RxFrame.end(), Out );
511
512 return true;
513}
