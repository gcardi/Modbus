76enum class ExceptionCode {
77 IllegalFunction = 1, // The function code received in the query
78 // is not an allowable action for the slave.
79 // If a Poll Program Complete command
80 // was issued, this code indicates that no
81 // program function preceded it.
82
83 IllegalDataAddress = 2, // The data address received in the query
84 // is not an allowable address for the
85 // slave.
86
87 IllegalDataValue = 3, // A value contained in the query data
88 // field is not an allowable value for the
89 // slave.
90
91 SlaveDeviceFailure = 4, // An unrecoverable error occurred while
92 // the slave was attempting to perform the
93 // requested action.
94
95 Acknowledge = 5, // The slave has accepted the request
96 // and is processing it, but a long duration
97 // of time will be required to do so. This
98 // response is returned to prevent a
99 // timeout error from occurring in the
100 // master. The master can next issue a
101 // Poll Program Complete message to
102 // determine if processing is completed.
103
104 SlaveDeviceBusy = 6, // The slave is engaged in processing a
105 // long�duration program command. The
106 // master should retransmit the message
107 // later when the slave is free.
108
109 NegativeAcknowledge = 7, // The slave cannot perform the program
110 // function received in the query. This
111 // code is returned for an unsuccessful
112 // programming request using function
113 // code 13 or 14 decimal. The master
114 // should request diagnostic or error
115 // information from the slave.
116
117 MemoryParityError = 8, // The slave attempted to read extended
118 // memory, but detected a parity error in
119 // the memory. The master can retry the
120 // request, but service may be required on
121 // the slave device.
122};
